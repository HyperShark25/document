Tkinter

Graphical User Interface

Using Tkinter library we can create a program that is used to calculate gram, ounce and pound.

from tkinter import *

def x():
    gram = float(y.get())*1000
    pound=float(y.get())*2.20462
    ounce=float(y.get())*35.274
    t1.delete(0.0,END)
    t2.delete(0.0,END)
    t3.delete(0.0,END)
    t1.insert(END,str(gram)+" gram")
    t2.insert(END,str(pound)+ " pound")
    t3.insert(END,str(ounce)+" ounce")
    e.delete(0,END)


win=Tk()
y=StringVar()
lab=Label(win, text="Kg")
lab.grid(row=0)
e=Entry(win, textvariable=y)
e.grid(row=0, column=1)
b=Button(win, text="Convert", command=x)
b.grid(row=0, column=2)
t1=Text(win, height=1, width=20)
t1.grid(row=1)
t2=Text(win, height=1, width=20)
t2.grid(row=1, column=1)
t3=Text(win, height=1, width=20)
t3.grid(row=1, column=2)
win.mainloop()

Rolling Dice game

Executing Functions

By importing a library in python called random, we can use it to create a random number made up totally by the computer.

import random


class Dice:
    def roll(self):
        first = random.randint(1, 6)
        second = random.randint(1, 6)
        return first, second


dice = Dice()
    print(dice.roll())
